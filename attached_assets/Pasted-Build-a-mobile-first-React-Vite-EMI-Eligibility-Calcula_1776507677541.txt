const RATE_DEFAULTS = {
  personal: 14,
  home: 8.5,
  lap: 11,
  business: 16
}

export function calcCCObligation(cc_outstanding) {
  return (cc_outstanding || 0) * 0.05
}

export function calcTotalExistingEMI(obligations) {
  return (
    (obligations.home_loan_emi || 0) +
    (obligations.personal_loan_emi || 0) +
    (obligations.business_loan_emi || 0) +
    (obligations.vehicle_loan_emi || 0)
  )
}

export function calcFOIR(profile) {
  // Base FOIR by employment type
  let foir = profile.employment_type === 'salaried' ? 0.65 : 0.55
  // Boost for premium employer categories
  if (profile.employment_type === 'salaried') {
    if (['mnc','listed','government_psu'].includes(profile.employer_category)) foir = 0.75
    else if (profile.employer_category === 'pvt_ltd') foir = 0.70
  }
  if (profile.employment_type === 'self_employed') {
    if (profile.itr_filed && profile.gst_registered) foir = 0.65
    else if (profile.itr_filed) foir = 0.60
  }
  // Reduce for risk
  if (profile.overdue_last_6months) foir -= 0.15
  if (profile.salary_mode === 'cash') foir -= 0.10
  return Math.min(0.80, Math.max(0.30, foir))
}

export function calcEligibility(profile, obligations, loanIntent) {
  const income = profile.employment_type === 'salaried'
    ? profile.monthly_net_income
    : profile.avg_monthly_bank_credit

  const totalExistingEMI = calcTotalExistingEMI(obligations)
  const ccObligation = calcCCObligation(obligations.cc_outstanding)
  const totalObligations = totalExistingEMI + ccObligation
  const foir = calcFOIR({ ...profile, ...obligations })
  const eligibleEMI = (income * foir) - totalObligations

  if (eligibleEMI <= 0) {
    return { eligible: false, eligibleEMI: 0, eligibleLoanAmount: 0,
             foirApplied: foir, totalObligations, surplusIncome: 0,
             riskCategory: 'high', leadScore: 10 }
  }

  const rate = RATE_DEFAULTS[loanIntent.loan_type] || 12
  const tenure = loanIntent.preferred_tenure_months || 60
  const monthlyRate = rate / 12 / 100
  const loanAmount = eligibleEMI * ((1 - Math.pow(1 + monthlyRate, -tenure)) / monthlyRate)

  // Lead score
  let score = 50
  if (profile.employment_type === 'salaried') {
    if (['mnc','listed','government_psu'].includes(profile.employer_category)) score += 15
    if ((profile.current_company_tenure_months || 0) >= 24) score += 10
    if (profile.salary_mode === 'bank_transfer') score += 10
    if (profile.pf_deducted) score += 5
  } else {
    if (profile.itr_filed) score += 15
    if (profile.gst_registered) score += 10
    if ((profile.business_vintage_years || 0) >= 3) score += 10
  }
  if (obligations.overdue_last_6months) score -= 25
  if (foir > 0.70) score -= 10
  score = Math.min(100, Math.max(0, score))

  const riskCategory = score >= 70 ? 'low' : score >= 40 ? 'medium' : 'high'

  return {
    eligible: true,
    eligibleEMI: Math.round(eligibleEMI),
    eligibleLoanAmount: Math.round(loanAmount),
    foirApplied: foir,
    totalObligations: Math.round(totalObligations),
    surplusIncome: Math.round(income - (income * foir)),
    riskCategory,
    leadScore: score,
    interestRateAssumed: rate
  }
}

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Mobile-first. Max width 480px, centered on desktop. 
2. All inputs min height 48px for easy tap.
3. Show validation errors only on field blur or form submit — not while typing.
4. Salaried fields must be completely hidden (not just disabled) when self_employed is selected and vice versa. Switching employment_type resets the hidden section's fields.
5. Live calculations on Screen 2 update on every keystroke.
6. Tenure options on Screen 3 must update immediately when loan_type changes.
7. Interest rate badge on Screen 3 updates immediately when loan_type changes.
8. If user navigates Back, re-populate fields from Zustand store (don't lose data).
9. Format all currency displays in Indian number system (₹1,23,45,678).
10. Disable the submit button and show a spinner while the Supabase insert is in progress.
11. Never show the result page if the user navigates directly to /result without completing previous screens — redirect to /.
12. All form state lives in Zustand, not in individual component state, so back navigation works.
